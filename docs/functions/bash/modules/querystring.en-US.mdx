The querystring module is imported.

```bash
#!/usr/bin/env import

import "querystring@1.3.0"

type 

handler() {
	http_response_json
	local path
	local query
	path="$(jq -r '.path' < "$1")"
	echo "{ 
		\"STS_API_KEY\": \"($query)\",
	}"
}
```
