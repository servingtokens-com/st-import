```bash
#!/usr/bin/env import

# Import script files to the `/tmp` directory
IMPORT_CACHE="/tmp"
# Log information related to `import` to stderr
IMPORT_DEBUG=1
# Force a fresh download of script files (like Shift + Reload in the browser)
IMPORT_RELOAD=1

build() {
    date > build-time.txt
}

handler() {
    eval "$(curl -sfLS https://import.sh)"
    type import
    import --version
    import assert
    type assert 
    assert --version
    echo "Build time: $(cat build-time.txt)"
	echo "Current time: $(date)"
}
```
